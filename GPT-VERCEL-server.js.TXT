require('dotenv').config();
const express = require('express');
const cors = require('cors');
const mongoose = require('mongoose');
const { generateText } = require('ai');

const app = express();

// שימוש ב-CORS וב-JSON
app.use(cors());
app.use(express.json());

// 🟢 קאש גלובלי לחיבור MongoDB (Serverless friendly)
let cached = global.mongoose;

if (!cached) {
    cached = global.mongoose = { conn: null, promise: null };
}

async function connectToDatabase(uri) {
    if (cached.conn) return cached.conn;

    if (!cached.promise) {
        console.log("Connecting to MongoDB...");
        cached.promise = mongoose.connect(uri)
            .then((mongoose) => {
                console.log("Connected to MongoDB!");
                return mongoose;
            })
            .catch((err) => {
                console.log("MongoDB connection error:", err);
                throw err;
            });
    }
    cached.conn = await cached.promise;
    return cached.conn;
}

// סכמת מתכון
const recipeSchema = new mongoose.Schema({
    title: String,
    ingredients: [String],
    instructions: String,
});

const RecipeModel = mongoose.models.Recipe || mongoose.model('Recipe', recipeSchema);

// 🌟 Route לבדיקה
app.get('/', (req, res) => {
    console.log("Hello World route hit!");
    res.send({ message: 'Hello World-11' });
});

// Route לקבלת כל המתכונים
app.get('/recipes', async (req, res) => {
    try {
        await connectToDatabase(process.env.DB_URL);
        const recipes = await RecipeModel.find();
        console.log("Fetched recipes successfully!");
        res.json(recipes);
    } catch (err) {
        console.log("Error fetching recipes:", err);
        res.status(500).json({ message: err.message });
    }
});

// Route להוספת מתכון חדש
app.post('/recipes', async (req, res) => {
    try {
        await connectToDatabase(process.env.DB_URL);
        const recipe = new RecipeModel({
            title: req.body.title,
            ingredients: req.body.ingredients,
            instructions: req.body.instructions
        });
        await recipe.save();
        console.log("Recipe created!");
        res.send({ message: 'Recipe created' });
    } catch (err) {
        console.log("Error creating recipe:", err);
        res.status(500).json({ message: err.message });
    }
});

// Route למחיקת מתכון
app.delete('/recipes/:id', async (req, res) => {
    try {
        await connectToDatabase(process.env.DB_URL);
        await RecipeModel.findByIdAndDelete(req.params.id);
        res.json({ message: 'Recipe deleted' });
    } catch (err) {
        console.log("Error deleting recipe:", err);
        res.status(500).json({ message: err.message });
    }
});

// Route ליצירת מתכון עם AI
app.post('/recipes/generate', async (req, res) => {
    try {
        await connectToDatabase(process.env.DB_URL);
        const { title, ingredients } = req.body;
        console.log("Generating recipe for:", title, ingredients);

        const { text } = await generateText({
            model: "anthropic/claude-sonnet-4.5",
            prompt: `Write a short recipe for "${title}" using these ingredients: ${ingredients}. Limit to 20 words.`
        });

        console.log("Generated text:", text);
        res.send({ recipe: text });
    } catch (err) {
        console.log("Error generating recipe:", err);
        res.status(500).json({ message: err.message });
    }
});

// שימוש בפורט מה-Environment (ל-Vercel) או פורט לוקלי
const PORT = process.env.PORT || 3000;
app.listen(PORT, () => {
    console.log(`Server is running on port ${PORT}`);
});

module.exports = app; // Vercel Serverless expects module.exports
